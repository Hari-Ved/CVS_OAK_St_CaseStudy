MASKING FUNCTION (Unity Catalog)
CREATE OR REPLACE FUNCTION oak_health.security.mask_phi(value STRING)
  RETURNS STRING
  RETURN CASE
    WHEN is_member('pii_readers') THEN value
    ELSE CONCAT('REDACTED-', SUBSTR(SHA2(value, 256), 1, 8))
  END;

ALTER TABLE oak_health.clinical.member_records
  ALTER COLUMN full_name      SET MASK oak_health.security.mask_phi;
  ALTER COLUMN date_of_birth  SET MASK oak_health.security.mask_phi;
  ALTER COLUMN email          SET MASK oak_health.security.mask_phi;

-- 1. Reconstruct what was accessed
SELECT user_identity.email, action_name,
  request_params.table_full_name,
  request_params.query_text,
  event_time
FROM system.access.audit
WHERE user_identity.email = '[email protected]'
  AND event_time BETWEEN '2025-03-13 08:00' AND '2025-03-13 09:30'
  AND action_name IN ('commandSubmit', 'getTable', 'readTable')
ORDER BY event_time DESC;
-- 2. Check PHI table access by non-authorized users (runs every 6h)
SELECT a.user_identity.email, COUNT(*) AS access_count
FROM system.access.audit a
WHERE a.request_params.table_full_name LIKE 'oak_health.clinical.%'
  AND event_time > DATEADD(hour, -6, CURRENT_TIMESTAMP)
  AND a.user_identity.email NOT IN (
    SELECT member FROM system.access.group_members
    WHERE group_name = 'pii_readers')
GROUP BY 1 HAVING access_count > 0;
-- Alert fires if any rows returned

    ELSE CONCAT('REDACTED-', SUBSTR(SHA2(value, 256), 1, 8))
